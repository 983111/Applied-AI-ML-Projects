01 Activation Functions

   

Each neuron applies a non-linear activation. We need both the forward computation and its exact derivative for backprop.

   
Sigmoid   σ(z) = 1 / (1 + e−z)        dσ/dz = σ(z)·(1 − σ(z)) ReLU       f(z) = max(0, z)             df/dz = 𝟙[z > 0] Softmax    sᵢ = eᶻⁱ / Σⱼ eᶻʲ          (numerically shifted)

def sigmoid(z):     """Numerically stable sigmoid."""     return 1.0 / (1.0 + np.exp(-np.clip(z, -500, 500))) def sigmoid_grad(z):     s = sigmoid(z); return s * (1 - s) def relu(z):     return np.maximum(0, z) def relu_grad(z): return (z > 0).astype(float) def softmax(z):     """Row-wise, shifted for stability."""     e = np.exp(z - z.max(axis=1, keepdims=True))     return e / e.sum(axis=1, keepdims=True) print(f"sigmoid(0)        = {sigmoid(np.array([[0.0]]))[0,0]:.4f}  (expect 0.5)") print(f"softmax([1,2,3])  = {np.round(softmax(np.array([[1.,2.,3.]])), 4)}  (sums to 1)")

02 Forward Pass

   

For each layer ℓ, compute the pre-activation Z then apply the activation to get A. Cache both tensors — backprop needs them.

   
Pre-activation   Z⁽ˡ⁾ = A⁽ˡ⁻¹⁾ · W⁽ˡ⁾ + b⁽ˡ⁾ Activation       A⁽ˡ⁾ = f⁽ˡ⁾(Z⁽ˡ⁾) He init          W ~ 𝒩(0, √(2/nᵢₙ))   ← prevents vanishing gradients with ReLU

class NeuralNetwork:     def __init__(self, layer_sizes, lr=0.01):         self.sizes = layer_sizes         self.lr    = lr         self.L     = len(layer_sizes) - 1         self.params = {}         for l in range(1, self.L + 1):             n_in, n_out = layer_sizes[l-1], layer_sizes[l]             self.params[f'W{l}'] = (                 np.random.randn(n_in, n_out) * np.sqrt(2.0 / n_in)  # He init             )             self.params[f'b{l}'] = np.zeros((1, n_out))     def forward(self, X):         cache = {'A0': X}         for l in range(1, self.L):          # hidden: ReLU             Z = cache[f'A{l-1}'] @ self.params[f'W{l}'] + self.params[f'b{l}']             cache[f'Z{l}'], cache[f'A{l}'] = Z, relu(Z)         l = self.L                           # output: softmax         Z = cache[f'A{l-1}'] @ self.params[f'W{l}'] + self.params[f'b{l}']         cache[f'Z{l}'], cache[f'A{l}'] = Z, softmax(Z)         self.cache = cache         return cache[f'A{self.L}'] nn = NeuralNetwork([2, 32, 32, 2], lr=0.05) print(f"Output shape : {nn.forward(np.random.randn(5,2)).shape}")

03 Backpropagation

   

Propagate the error signal from output → input via the chain rule. The softmax + cross-entropy output delta has an elegant closed form.

   
Loss            ℒ = −(1/m) Σ yᵢ · log(ŷᵢ)         cross-entropy Output delta    δᴸ = Â − Y                          softmax + CE combined Hidden delta    δˡ = (δˡ⁺¹ · Wˡ⁺¹ᵀ) ⊙ ReLU′(Zˡ)   chain rule Weight grad     ∂ℒ/∂Wˡ = Aˡ⁻¹ᵀ · δˡ / m Bias grad       ∂ℒ/∂bˡ = Σᵢ δˡᵢ / m
   
◈ The combined gradient of softmax + cross-entropy simplifies to ŷ − y. Computing the full softmax Jacobian is unnecessary.

def cross_entropy(y_pred, y_true):     m = y_true.shape[0]     return -np.log(np.clip(y_pred[np.arange(m), y_true], 1e-9, 1)).mean() def backward(self, y_true):     grads, m = {}, y_true.shape[0]     cache     = self.cache     # ── δᴸ = Â − Y  (softmax + CE combined gradient) ────────────     dZ = self.cache[f'A{self.L}'].copy()     dZ[np.arange(m), y_true] -= 1     dZ /= m     # ── Back-propagate layer by layer ────────────────────────────     for l in reversed(range(1, self.L + 1)):         grads[f'dW{l}'] = cache[f'A{l-1}'].T @ dZ         grads[f'db{l}'] = dZ.sum(axis=0, keepdims=True)         if l > 1:             dZ = (dZ @ self.params[f'W{l}'].T) * relu_grad(cache[f'Z{l-1}'])     return grads import types nn.backward = types.MethodType(backward, nn) # Verify gradient shapes y_test = np.array([0,1,0,1,0]) _ = nn.forward(np.random.randn(5,2)) grads = nn.backward(y_test) print("\n".join(f"  ∂L/∂{k:3s}: shape={str(v.shape):12s}  ‖∇‖={np.abs(v).mean():.2e}"                  for k,v in grads.items()))

04 Gradient Descent Update

   

Apply the standard SGD rule to every weight matrix and bias vector, then train with mini-batches.

   
Update rule   Wˡ ← Wˡ − η · ∂ℒ/∂Wˡ               bˡ ← bˡ − η · ∂ℒ/∂bˡ
   
◉ Mini-batch SGD: shuffle the dataset each epoch, split into batches of 32, perform one forward + backward + update pass per batch.

def sgd_update(self, grads):     for l in range(1, self.L + 1):         self.params[f'W{l}'] -= self.lr * grads[f'dW{l}']         self.params[f'b{l}'] -= self.lr * grads[f'db{l}'] nn.sgd_update = types.MethodType(sgd_update, nn) def train(model, X, y, epochs=500, batch_size=32):     loss_hist, acc_hist = [], []     m = X.shape[0]     for epoch in range(epochs):         idx = np.random.permutation(m)         X_s, y_s = X[idx], y[idx]         for s in range(0, m, batch_size):             Xb, yb = X_s[s:s+batch_size], y_s[s:s+batch_size]             model.forward(Xb)             model.sgd_update(model.backward(yb))         pred  = model.forward(X)         loss  = cross_entropy(pred, y)         acc   = (pred.argmax(axis=1) == y).mean()         loss_hist.append(loss); acc_hist.append(acc)         if (epoch + 1) % 100 == 0:             bar = '▓' * int(acc * 20) + '░' * (20 - int(acc * 20))             print(f"  {epoch+1:4d}/{epochs}  ℒ={loss:.4f}  acc={acc*100:5.1f}%  {bar}")     return loss_hist, acc_hist # ── Generate 2-class spiral dataset ────────────────────────────── def make_spiral(n=400, seed=42):     np.random.seed(seed); X, y = [], []     for c in range(2):         for i in range(n//2):             r = i/(n//2) * 0.9 + 0.05             t = (i/(n//2)) * 4 + c*np.pi + np.random.randn()*0.25             X.append([r*np.cos(t), r*np.sin(t)]); y.append(c)     X = np.array(X, dtype=float); X /= np.abs(X).max()     return X, np.array(y) X_train, y_train = make_spiral(400) nn = NeuralNetwork([2, 32, 32, 2], lr=0.05) nn.backward   = types.MethodType(backward, nn) nn.sgd_update = types.MethodType(sgd_update, nn) loss_hist, acc_hist = train(nn, X_train, y_train, epochs=500, batch_size=32)
